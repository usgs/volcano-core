1   package gov.usgs.volcanoes.core.util;
2   
3   import gov.usgs.volcanoes.core.contrib.HashCodeUtil;
4   
5   /**
6    * Container for two objects, possible different types.
7    * 
8    * @author Dan Cervelli
9    */
10  public class Pair<O1, O2> {
11    public O1 item1;
12    public O2 item2;
13  
14    /**
15     * Constructor.
16     * 
17     * @param o1 first object
18     * @param o2 second object
19     */
20    public Pair(O1 o1, O2 o2) {
21      item1 = o1;
22      item2 = o2;
23    }
24  
25    /**
26     * Yields true if otherPair same as this.
27     * 
28     * @param other pair being compared to this
29     * @return result of comparison
30     */
31    @Override
32    public boolean equals(Object other) {
33      if (!(other instanceof Pair<?, ?>)) {
34        return false;
35      } else {
36        Pair<?, ?> otherPair = (Pair<?, ?>) other;
37        return item1.equals(otherPair.item1) && item2.equals(otherPair.item2);
38      }
39    }
40  
41    /**
42     * Provide hashCode.
43     */
44    @Override
45    public int hashCode() {
46      int result = HashCodeUtil.SEED;
47      result = HashCodeUtil.hash(result, item1);
48      result = HashCodeUtil.hash(result, item2);
49  
50      return result;
51    }
52  
53    /**
54     * String representation of this.
55     * 
56     * @return String representation of this 
57     */
58    public String toString() {
59      return item1.toString() + "&" + item2.toString();
60    }
61  }
