72    /** 
73     * Sets the location of the frame in the plot to be no bigger than wxh
74     * @param x the x-pixel location
75     * @param y the y-pixel location
76     * @param w the width in pixels
77     * @param h the height in pixels
78     */
79    public void setLocationByMaxBounds(int x, int y, int w, int h) {
80      double[] extents = range.getProjectedExtents(projection);
81      double aspect = (extents[3] - extents[2]) / (extents[1] - extents[0]);
82  
83      int height = (int) ((double) w * aspect);
84      // System.out.println("try " + w + "x" + height);
85      if (height <= h) {
86        setLocation(x, y, w, height);
87        return;
88      }
89  
90      int width = (int) ((double) h * 1 / aspect);
91      // System.out.println("try " + width + "x" + h);
92      if (width <= w) {
93        int ofs = w - width;
94        setLocation(x + ofs / 2, y, width, h);
95        return;
96      }
97  
98      // System.out.println("fallback (out of aspect): " + w + "x" + h);
99      setLocation(x, y, w, h);
100   }

161   /**  
162    * Shortcut for createGraticule(pts, true, true, createLabels, createLabels, Color.BLACK)
163    * @param pts # of ticks
164    * @param xTickMarks if true, tick marks on X axis are rendered
165    * @param yTickMarks if true, tick marks on Y axis are rendered
166    * @param xTickValues if true, tick values on X axis are rendered
167    * @param yTickValues if true, tick values on Y axis are rendered
168    * @param labelColor color of label text
169    */
170   public void createGraticule(int pts, boolean xTickMarks, boolean yTickMarks, boolean xTickValues,
171       boolean yTickValues, Color labelColor) {
172     double[] gridX =
173         SmartTick.autoTick(range.getWest(), range.getWest() + range.getLonRange(), pts, true, true);
174     double[] gridY = SmartTick.autoTick(range.getSouth(), range.getNorth(), pts, true, true);
175     double[] gridXreal = null;
176     double[] gridYreal = null;
177     if (xTickMarks) {
178       gridXreal = gridX;
179     } else {
180       gridXreal = new double[2];
181       gridXreal[0] = range.getWest();
182       gridXreal[1] = range.getWest() + range.getLonRange();
183     }
184     if (yTickMarks) {
185       gridYreal = gridY;
186     } else {
187       gridYreal = new double[2];
188       gridYreal[0] = range.getSouth();
189       gridYreal[1] = range.getNorth();
190     }
191     graticule = new LineData(gridXreal, gridYreal, false);
192     graticule.applyProjection(projection);
193     labels = new ArrayList<Renderer>();
194     if (xTickValues && (gridX != null)) {
195       for (int i = 1; i < gridX.length - 1; i++) {
196         double nx = GeoRange.normalize(gridX[i]);
197         Point2D.Double pt = new Point2D.Double(nx, gridY[0]);
198         pt = projection.forward(pt);
199         TextRenderer tr = new TextRenderer(pt.x, pt.y, numberFormat.format(nx));
200         tr.transformer = this;
201         tr.horizJustification = TextRenderer.CENTER;
202         tr.vertJustification = TextRenderer.TOP;
203         tr.color = labelColor;
204         labels.add(tr);
205       }
206     }
207     if (yTickValues && (gridY != null)) {
208       for (int i = 1; i < gridY.length - 1; i++) {
209         Point2D.Double pt = new Point2D.Double(gridX[0], gridY[i]);
210         pt = projection.forward(pt);
211         TextRenderer tr = new TextRenderer(pt.x, pt.y, numberFormat.format(gridY[i]));
212         tr.transformer = this;
213         tr.xBump = -2;
214         tr.horizJustification = TextRenderer.RIGHT;
215         tr.vertJustification = TextRenderer.CENTER;
216         tr.color = labelColor;
217         labels.add(tr);
218       }
219     }
220   }

244   /**  
245    * Yield true iff bb intersects any element of boxes
246    * @param boxes list of boxes to compare bb against
247    * @param bb a Rectangle
248    * @return true iff bb intersects any element of boxes
249    */
250   private boolean boxOverlaps(List<Rectangle> boxes, Rectangle bb) {
251     for (Iterator<Rectangle> it2 = boxes.iterator(); it2.hasNext();) {
252       Rectangle rect = it2.next();
253       if (rect.intersects(bb))
254         return false;
255     }
256 
257     return true;
258   }

260   /**  
261    * Yield location where to put a label, null if nowhere
262    * @param boxes list of boxes to compare bb against
263    * @param bb a Rectangle
264    * @return Point where label can go; null if nowhere
265    */
266   private Point allowLabel(List<Rectangle> boxes, Rectangle bb) {
267     if (boxOverlaps(boxes, bb))
268       return new Point(0, 0);
269 
270     bb.translate(-bb.width - 7, 0);
271     if (boxOverlaps(boxes, bb))
272       return new Point(-bb.width - 7, 0);
273 
274     bb.translate(0, bb.height / 2);
275     if (boxOverlaps(boxes, bb))
276       return new Point(-bb.width - 7, bb.height / 2 + 3);
277 
278     bb.translate(bb.width + 7, 0);
279     if (boxOverlaps(boxes, bb))
280       return new Point(0, bb.height / 2 + 3);
281 
282     return null;
283   }

285   /**
286    * Render labels om map
287    * @param g the graphics object upon which to render
288    */
289   public void renderGeoLabels(Graphics2D g) {
290     ArrayList<Rectangle> boxes = new ArrayList<Rectangle>();
291 
292     if (geoLabelSet != null) {
293       List<GeoLabel> gls = geoLabelSet.getGeoLabels();
294       for (GeoLabel gl : gls) {
295         if (!range.contains(gl.location))
296           continue;
297 
298         Point2D.Double pt = projection.forward(gl.location);
299 
300         if (gl.shadow) {
301           ShadowedTextRenderer str = new ShadowedTextRenderer(pt.x, pt.y, gl.text);
302           str.xBump = 4;
303           str.yBump = -1;
304           str.color = gl.color;
305           str.font = gl.font;
306           str.transformer = this;
307           Rectangle bb = str.getBoundingBox(g);
308           Point offset = allowLabel(boxes, bb);
309           if (offset != null) {
310             str.xBump += offset.x;
311             str.yBump += offset.y;
312             boxes.add(bb);
313             str.render(g);
314           }
315         } else {
316           TextRenderer tr = new TextRenderer(pt.x, pt.y, gl.text);
317           tr.font = gl.font;
318           tr.xBump = 4;
319           tr.yBump = -1;
320           tr.color = gl.color;
321           tr.transformer = this;
322 
323           Rectangle bb = tr.getBoundingBox(g);
324           Point offset = allowLabel(boxes, bb);
325           if (offset != null) {
326             tr.xBump += offset.x;
327             tr.yBump += offset.y;
328             boxes.add(bb);
329             tr.render(g);
330           }
331         }
332 
333 
334         if (gl.marker != null) {
335           gl.marker.transformer = this;
336           gl.marker.x = pt.x;
337           gl.marker.y = pt.y;
338           gl.marker.render(g);
339         }
340 
341 
342       }
343     }

363   /**
364    * Create renderer for scale
365    * @param f factor
366    * @param x x position
367    * @param y y position
368    * @param color color
369    */
370   public void createScaleRenderer(final double f, final int x, final int y, final Color color) {
371     scaleRenderer = new Renderer() {
372       public void render(Graphics2D g) {
373         Font origFont = g.getFont();
374         if (color != null)
375           g.setColor(color);
376         g.setFont(new Font("Arial", Font.PLAIN, 10));
377         g.setStroke(new BasicStroke(2.0f));
378         double scale = getScale() / 1000 * f;
379         Line2D.Double line = new Line2D.Double();
380         double len = 0;
381         int index = -1;
382         double cur = 1E300;
383         for (int i = 0; i < scales.length; i++) {
384           double s = 1.0 / scale * scales[i];
385           double d = Math.abs(s - 150);
386           if (d < cur) {
387             len = s;
388             index = i;
389             cur = d;
390           }
391         }
392         // line.setLine(graphX, graphY + graphHeight + 32, graphX + len, graphY + graphHeight + 32);
393         // line.setLine(graphX, graphY + graphHeight + 19, graphX + len, graphY + graphHeight + 19);
394         line.setLine(x, y, x + len, y);
395         g.draw(line);
396         // g.drawString(scales[index] + " km", (float)graphX, (float)(graphY + graphHeight + 30));
397         g.drawString(scales[index] + " km", (float) x, (float) (y - 2));
398         g.setFont(origFont);
399       }
400     };
401   }

403   /**
404    * Render map
405    * @param g the graphics object upon which to render
406    */
407   public void render(Graphics2D g) {
408     super.render(g);
409     // CodeTimer ct = new CodeTimer("MapRenderer.render");
410     double[] extents = range.getProjectedExtents(projection);
411     // System.out.printf("mr extents: %f, %f, %f, %f\n", extents[0], extents[1], extents[2],
412     // extents[3]);
413 
414     LineDataRenderer ldr = null;
415     LineDataRenderer sdr = null;
416 
417     if (mapImage != null) {
418       RenderedImageDataRenderer mapImageRenderer = new RenderedImageDataRenderer(mapImage);
419       mapImageRenderer.setLocation(this);
420       mapImageRenderer.setExtents(extents[0], extents[1], extents[2], extents[3]);
421       mapImageRenderer.setDataExtents(extents[0], extents[1], extents[2], extents[3]);
422       mapImageRenderer.render(g);
423     }
424     // ct.mark("mapImage");
425     if (lineDatas != null) {
426       for (LineData ld : lineDatas) {
427         ld.applyProjection(projection);
428         LineDataRenderer lldr = new LineDataRenderer(ld);
429         lldr.setAntiAlias(true);
430         lldr.setLocation(this);
431         lldr.setExtents(extents[0], extents[1], extents[2], extents[3]);
432         lldr.render(g);
433       }
434     }
435     // ct.mark("lineData");
436 
437     if (userDefined != null) {
438       for (LineData ld : userDefined) {
439         sdr = new LineDataRenderer(ld);
440         sdr.color = ld.color;
441         sdr.stroke = ld.stroke;
442         sdr.setLocation(this);
443         sdr.setExtents(extents[0], extents[1], extents[2], extents[3]);
444         sdr.render(g);
445       }
446     }
447 
448     if (box != null) {
449       ldr = new LineDataRenderer(box);
450       ldr.setLocation(this);
451       ldr.setExtents(extents[0], extents[1], extents[2], extents[3]);
452       // ldr.setPaint(NO_BACKGROUND_PAINT);
453       // ldr.render(g);
454       ldr.render(g);
455     }
456     // ct.mark("box");
457 
458     if (graticule != null) {
459       LineDataRenderer gldr = new LineDataRenderer(graticule);
460       gldr.setAntiAlias(true);
461       gldr.setLocation(this);
462       gldr.setExtents(extents[0], extents[1], extents[2], extents[3]);
463       gldr.stroke = LineRenderer.DASHED_STROKE;
464       gldr.color = Color.DARK_GRAY;
465       gldr.render(g);
466     }
467     // ct.mark("graticule");
468 
469     if (labels != null) {
470       for (Renderer label : labels)
471         label.render(g);
472     }
473     // ct.mark("labels");
474 
475     if (scaleRenderer != null)
476       scaleRenderer.render(g);
477 
478     if (geoLabelSet != null)
479       renderGeoLabels(g);
480 
481     // ct.mark("geolabels");
482     // g.setColor(Color.RED);
483     // g.drawRect(graphX, graphY, getGraphWidth(), getGraphHeight());
484     // g.drawLine(graphX, graphY + getGraphHeight() / 2, graphX + getGraphWidth(), graphY +
485     // getGraphHeight() / 2);
486     // g.drawLine(graphX + getGraphWidth() / 2, graphY, graphX + getGraphWidth() / 2, graphY +
487     // getGraphHeight());
488     // ct.stop();

491   /**
492    * Main method
493    * @param args command line arguments
494    */
495   public static void main(String[] args) {
496     GeoImageSet images = new GeoImageSet(args[0]);
497     GeoLabelSet labels = new GeoLabelSet(args[1]);
498     GeoRange range = new GeoRange(Double.parseDouble(args[2]), Double.parseDouble(args[3]),
499         Double.parseDouble(args[4]), Double.parseDouble(args[5]));
500 
501     Plot plot = new Plot();
502     plot.setBackgroundColor(Color.white);
503     plot.setSize(1200, 900);
504 
505     TransverseMercator proj = new TransverseMercator();
506     Point2D.Double origin = range.getCenter();
507     proj.setup(origin, 0, 0);
508 
509     MapRenderer mr = new MapRenderer(range, proj);
510     mr.setLocation(50, 50, 800);
511     mr.setGeoLabelSet(labels.getSubset(range));
512     RenderedImage ri = images.getMapBackground(proj, range, 800);
513     mr.setMapImage(ri);
514     mr.createBox(8);
515     mr.createGraticule(8, true);
516     mr.getDefaultTranslation(plot.getHeight());
517     plot.addRenderer(mr);
518     plot.quickShow();
519   }
