7   package gov.usgs.volcanoes.core;

10  import org.apache.log4j.Level;
11  import org.apache.log4j.Logger;
12  import org.apache.log4j.PatternLayout;
13  import org.apache.log4j.RollingFileAppender;
14  
15  import java.io.IOException;
16  
17  /**
18   * A utility class to handle implementation-specific logging configuration. Most classes will not
19   * need to configure the logger. Most classes that do need to configure the logger will do so
20   * through a properties file. I'm here for those few remaining classes.
21   * 
22   * @author Tom Parker
23   *
24   */
25  public class Log {
26    private static String LOG_PATTERN = "%d{yyyy-MM-dd hh:mm:ss} %5p - %m (%F:%L)%n";
27  
28    /**
29     * Add a rolling file appender to the root logger.
30     * 
31     * @param name name of file
32     * @throws IOException when file cannot be created or modified
33     */
34    public static void addFileAppender(String name) throws IOException {
35      PatternLayout layout = new PatternLayout(LOG_PATTERN);
36      RollingFileAppender fileAppender = new RollingFileAppender(layout, name);
37      fileAppender.setMaxFileSize("1MB");
38      fileAppender.setMaxBackupIndex(2);
39      Logger.getRootLogger().addAppender(fileAppender);
40    }
41  
42    /**
43     * Set root logger level.
44     * 
45     * @param level minimum logging level
46     */
47    public static void setLevel(Level level) {
48      Logger.getRootLogger().setLevel(level);
49    }
50  }
