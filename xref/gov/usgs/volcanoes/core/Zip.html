9   import java.util.ArrayList;
10  import java.util.zip.Deflater;
11  import java.util.zip.Inflater;

13  /**
14   * Utility classes for working with the JDK <code>Inflator</code>/
15   * <code>Deflator</code> implementation.
16   *
17   * @author Dan Cervelli
18   *
19   */
20  public class Zip {
21    /**
22     * Compresses an array of bytes using the JDK <code>Inflator</code>/
23     * <code>Deflator</code> implementation set for maximum speed.
24     * Compression ratios comparable to gzip are attained.
25     *
26     * @param bytes the array of bytes
27     * @return the compressed array of bytes
28     */
29    public static byte[] compress(byte[] bytes) {
30      return compress(bytes, Deflater.BEST_SPEED);
31    }
32  
33    /**
34     * Compresses an array of bytes using the JDK <code>Inflator</code>/
35     * <code>Deflator</code> implementation. Compression ratios comparable to
36     * gzip are attained.
37     *
38     * @param bytes the array of bytes
39     * @param level the compression level (1[least]-9[most])
40     * @return the compressed array of bytes
41     */
42    public static byte[] compress(byte[] bytes, int level) {
43      return compress(bytes, level, 0, bytes.length);
44    }
45  
46    /**
47     * Compresses an array of bytes using the JDK <code>Inflator</code>/
48     * <code>Deflator</code> implementation. Compression ratios comparable to
49     * gzip are attained.
50     *
51     * @param bytes the array of bytes
52     * @param level the compression level (1[least]-9[most])
53     * @param ofs number of first byte in array to process
54     * @param len length of processed array zone
55     * @return the compressed array of bytes
56     */
57    public static byte[] compress(byte[] bytes, int level, int ofs, int len) {
58      final Deflater deflater = new Deflater(level);
59      deflater.setInput(bytes, ofs, len);
60      deflater.finish();
61      final ArrayList<byte[]> list = new ArrayList<byte[]>(2);
62      boolean done = false;
63      int compSize = 0;
64      // must allow for the compressed size to be larger than the original size
65      while (!done) {
66        final byte[] compBuf = new byte[bytes.length];
67        compSize = deflater.deflate(compBuf);
68        if (deflater.finished()) {
69          done = true;
70        }
71        list.add(compBuf);
72      }
73      final int total = (list.size() - 1) * bytes.length + compSize;
74      final byte[] finalBuf = new byte[total];
75      int totalLen = 0;
76      for (int i = 0; i < list.size() - 1; i++) {
77        System.arraycopy(list.get(i), 0, finalBuf, totalLen, bytes.length);
78        totalLen += bytes.length;
79      }
80      System.arraycopy(list.get(list.size() - 1), 0, finalBuf, totalLen, compSize);
81      return finalBuf;
82    }
83  
84    /**
85     * Decompresses an array of bytes compressed via the <code>Util.compress()</code>
86     * methods. Uses a bufferSize of 64K.
87     *
88     * @param bytes the compressed bytes
89     * @return the decompressed array of bytes
90     */
91    public static byte[] decompress(byte[] bytes) {
92      return decompress(bytes, 65536);
93    }
94  
95    /**
96     * Decompresses an array of bytes compressed via the <code>Util.compress()</code>
97     * methods. Use a bufferSize slightly larger than the expected size of the
98     * decompressed data for maximum efficiency.
99     *
100    * @param bytes the compressed bytes
101    * @param bufferSize the decompression buffer size
102    * @return the decompressed array of bytes
103    */
104   public static byte[] decompress(byte[] bytes, int bufferSize) {
105     try {
106       final Inflater inflater = new Inflater();
107       inflater.setInput(bytes);
108       boolean done = false;
109 
110       final ArrayList<byte[]> list = new ArrayList<byte[]>(10);
111       int numBytes = 0;
112       while (!done) {
113         final byte[] buffer = new byte[bufferSize];
114         numBytes = inflater.inflate(buffer);
115         if (inflater.finished()) {
116           done = true;
117         }
118         list.add(buffer);
119       }
120       inflater.end();
121       final int total = (list.size() - 1) * bufferSize + numBytes;
122       final byte[] finalBuf = new byte[total];
123       int totalLen = 0;
124       for (int i = 0; i < list.size() - 1; i++) {
125         System.arraycopy(list.get(i), 0, finalBuf, totalLen, bufferSize);
126         totalLen += bufferSize;
127       }
128       System.arraycopy(list.get(list.size() - 1), 0, finalBuf, totalLen, numBytes);
129       return finalBuf;
130     } catch (final Exception e) {
131       e.printStackTrace();
132     }
133     return null;
134   }
