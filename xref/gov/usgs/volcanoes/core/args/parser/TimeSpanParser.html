6   package gov.usgs.volcanoes.core.args.parser;
7   
8   import com.martiansoftware.jsap.ParseException;
9   import com.martiansoftware.jsap.StringParser;
10  
11  import gov.usgs.volcanoes.core.time.Time;
12  import gov.usgs.volcanoes.core.time.TimeSpan;
13  
14  import java.text.SimpleDateFormat;
15  import java.util.Locale;
16  import java.util.TimeZone;
17  
18  /**
19   * Parse a time span from a command line argument.
20   * 
21   * @author Tom Parker
22   */
23  public class TimeSpanParser extends StringParser {
24  
25    private SimpleDateFormat format;
26  
27    /**
28     * Constructor.
29     * 
30     * @param inputFormat Format string suitable for feeding to SimpleDataFormat
31     */
32    public TimeSpanParser(String inputFormat) {
33      format = new SimpleDateFormat(inputFormat, Locale.ENGLISH);
34      format.setTimeZone(TimeZone.getTimeZone("UTC"));
35    }
36  
37    @Override
38    public TimeSpan parse(String arg) throws ParseException {
39  
40      int seperatorIdx = arg.lastIndexOf('-');
41  
42      String startString;
43      String endString;
44      if (seperatorIdx > 0) {
45        startString = arg.substring(0, seperatorIdx);
46        endString = arg.substring(seperatorIdx + 1);
47      } else {
48        startString = arg;
49        endString = "";
50      }
51  
52      long endTime;
53      long startTime;
54  
55      if (endString.length() == 0 || "now".equalsIgnoreCase(endString)) {
56        endTime = System.currentTimeMillis();
57      } else {
58        try {
59          endTime = format.parse(endString).getTime();
60        } catch (java.text.ParseException ex) {
61          throw new ParseException("Unable to convert  end '" + endString + "' to a time.");
62        }
63      }
64  
65      try {
66        startTime = format.parse(startString).getTime();
67      } catch (java.text.ParseException ex) {
68  
69        double start = Time.getRelativeTime(startString);
70        if (!Double.isNaN(start)) {
71          startTime = endTime - (long) (start * 1000);
72  
73        } else {
74          throw new ParseException("Unable to convert  start '" + startString + "' to a time.");
75        }
76      }
77  
78      return new TimeSpan(startTime, endTime);
79    }
80  }
