6   package gov.usgs.volcanoes.core.args.decorator;
7   
8   import com.martiansoftware.jsap.JSAP;
9   import com.martiansoftware.jsap.UnflaggedOption;
10  
11  import gov.usgs.volcanoes.core.args.ArgsDecorator;
12  import gov.usgs.volcanoes.core.args.ArgumentException;
13  import gov.usgs.volcanoes.core.args.Arguments;
14  
15  /**
16   * Gather a config filename from the command line.
17   *
18   * <p>The filename is taken from the first available unflagged option and placed in
19   * "config-filename". Be careful of the order that Args are added.
20   *
21   * @author Tom Parker
22   */
23  public class ConfigFileArg extends ArgsDecorator {
24  
25    /**
26     * Constructs a ConfigFileArg and adds its parameter to the list.
27     *
28     * @param defaultFileName The value to assign if the parameter is not given on the command line
29     * @param nextArg The next Arguments object in the chain
30     * @throws ArgumentException if new Parameter cannot be created. Not sure why this would happen
31     */
32    public ConfigFileArg(final String defaultFileName, final Arguments nextArg)
33        throws ArgumentException {
34      super(nextArg);
35  
36      registerParameter(new UnflaggedOption("config-filename", JSAP.STRING_PARSER, defaultFileName,
37          JSAP.NOT_REQUIRED, JSAP.NOT_GREEDY, "The config file name."));
38    }
39  }
