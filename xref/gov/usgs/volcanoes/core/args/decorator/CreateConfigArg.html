6   package gov.usgs.volcanoes.core.args.decorator;

8   import com.martiansoftware.jsap.JSAPException;
9   import com.martiansoftware.jsap.JSAPResult;
10  import com.martiansoftware.jsap.Switch;
11  
12  import gov.usgs.volcanoes.core.args.ArgsDecorator;
13  import gov.usgs.volcanoes.core.args.ArgumentException;
14  import gov.usgs.volcanoes.core.args.Arguments;
15  
16  import java.io.File;
17  import java.io.IOException;
18  import java.io.InputStream;
19  import java.nio.file.Files;
20  import java.nio.file.Path;
21  
22  /**
23   * Create an example config file.
24   *
25   * <p>CreateConfigArg implies ConfigFileArg
26   *
27   * @author Tom Parker
28   */
29  public class CreateConfigArg extends ArgsDecorator {
30  
31    private final String exampleConfigFile;
32  
33    /**
34     * Construct a CreateConfigArg adding its Parameter to the list.
35     *
36     * @param exampleConfigFile String resource of example configfile
37     * @param nextArg The next Argument in the list
38     * @throws ArgumentException if parameter is already registered or cannot be added.
39     */
40    public CreateConfigArg(final String exampleConfigFile, Arguments nextArg)
41        throws ArgumentException {
42      super(nextArg);
43      this.exampleConfigFile = exampleConfigFile;
44  
45      nextArg.registerParameter(new Switch("create-config", 'c', "create-config",
46          "Create an example config file in the curent working directory."));
47  
48      if (nextArg.getById("config-filename") == null) {
49        throw new ArgumentException("CreateConfigArg relies on ConfigFileArg. Please wrap it first.");
50      }
51    }
52  
53    @Override
54    public JSAPResult parse(final String[] args) throws ArgumentException {
55      final JSAPResult jsap = super.parse(args);
56      final String configFileName = jsap.getString("config-filename");
57      if (jsap.getBoolean("create-config")) {
58        final InputStream is =
59            ClassLoader.getSystemClassLoader().getResourceAsStream(exampleConfigFile);
60        final Path defaultPath = new File(configFileName).toPath();
61        try {
62          Files.copy(is, defaultPath);
63        } catch (IOException e) {
64          throw new ArgumentException(e);
65        }
66      }
67      return jsap;
68    }
69  }
