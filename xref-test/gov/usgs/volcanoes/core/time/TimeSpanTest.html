15      assertTrue("30s".equals(ts.span()));
16  
17      ts = new TimeSpan(30 * TimeSpan.SECOND, 0);
18      assertTrue("-30s".equals(ts.span()));
19  
20      long time =
21          TimeSpan.DAY + (2 * TimeSpan.HOUR) + (30 * TimeSpan.MINUTE) + (TimeSpan.SECOND) + 531;
22      ts = new TimeSpan(0, time);
23      assertTrue("1d 2h 30m 1.531s".equals(ts.span()));
24    }

37      TimeSpan ts = TimeSpan.parse("-24h,20171102000000");
